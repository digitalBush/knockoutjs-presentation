Observable
  • Subscribe with a callback.
  • Get notified when value changes.

Observable Example
//Construction            
var viewModel={
    Name:ko.observable("Josh Bush")
};

//Set Value
viewModel.Name("digitalBush");

//Get Value
var who = viewModel.Name();

//Manual Subscription
viewModel.Name.subscribe(function(val) {
    alert("I can haz new name: " + val);
});
Observable Array
  • Subscribe with a callback.
  • Get notified when array contents change.
  • Exposes methods from underlying array like push, pop, shift, unshift, etc.

Observable Array Example

var viewModel={
    Tags:ko.observableArray([
        "Javascript",
        "Templating",

Computed Observable
  • Define a function to compose other observable value(s)
  • Dependent observable subscriptions autowired

Computed Observable Example
var viewModel={
    Subtotal:ko.observable(5)
};

viewModel.Tax=ko.computed(function(){
    return this.Subtotal()*0.0925;
},viewModel);
You can write to them too!
viewModel.Total=ko.computed({
    read: function () {
        return this.Subtotal() + this.Tax(); 
    },
    write: function (value) {
        this.Subtotal(value/1.0925)
    },
    owner: viewModel
});

Custom Binding
ko.bindingHandlers.mask = {
    init: function(elm, valueAccessor, allBindingsAccessor) {
        var val = ko.utils.unwrapObservable(valueAccessor());
        var allBindings = allBindingsAccessor();
        var mask = allBindings.maskValue; 
        $(elm).val(val).mask(mask)
        .change(function(val){
            valueAccessor()($(this).mask());
        });
    },
    update: function(elm, valueAccessor, allBindingsAccessor) {
        var val = ko.utils.unwrapObservable(valueAccessor());
        $(elm).val(val).trigger("remask");
    }
};

The Knockout Version
var viewModel={
    Phones:ko.observableArray([]),
    add:function(){
        this.Phones.push({Number:ko.observable()});
    },
    save:function(){
        var json=ko.toJSON(
            ko.utils.arrayMap(
                this.Phones(),
                function(item){return item.Number();}
            ));
        alert(json);
    },
    remove:function(phone){
        viewModel.Phones.remove(phone);
    }
};
        
<label>Phone Numbers:</label>
<ul data-bind="foreach:Phones">
    <li>
        <input data-bind="mask:Number,maskValue:'(999) 999-9999'" />
        <button data-bind='click:$root.remove'>
            <i class="icon-remove"></i>
        </button>
    </li>
</ul>
<button data-bind="click:add">Add Phone</button>
<p>
    <button data-bind="click:save">Save</button>
</p>
